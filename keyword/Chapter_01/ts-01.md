# 반환값으로, 설정한 타입과 할당한 변수의 타입이 맞지 않으면, 어떻게 되는지 아래에 작성해주세요!

## string
### 성공
```
let userName: string = "서현";
```
### 실패
```
let userName: string = 123;
```

## number
### 성공
```
let age: number = 25;
```
### 실패
```
let age: number = "25";
```

## boolean
### 성공
```
let isLogin: boolean = true;
```
### 실패
```
let isLogin: boolean = "true";
```

## null
### 성공
```
let data: null = null;
```
### 실패
```
let data: null = undefined;
```
## undefined
### 성공
```
let value: undefined = undefined;
```
### 실패
```
let value: undefined = null;
```

## symbol
### 성공
```
let sym: symbol = Symbol("id");
```
### 실패
```
let sym: symbol = "id";
```
## bigint
### 성공
```
let bigNumber: bigint = 12345678901234567890n;
```
### 실패
```
let bigNumber: bigint = 12345678901234567890;
```

## object
### 성공
```
let user: object = { name: "서현", age: 23 };
```
### 실패
```
let user: object = "서현";
```
